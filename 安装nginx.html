1. 安装环境
yum install -y pcre pcre-devel zlib zlib-devel openssl openssl-devel gcc
2. 解压nginx
tar -zxvf nginx****
3. 编译安装三部曲
./configure --prefix=/usr/local/nginx --pid-path=/var/run/nginx/nginx.pid --lock-path=/var/lock/nginx.lock --error-log-path=/var/log/nginx/error.log --http-log-path=/var/log/nginx/access.log --with-http_gzip_static_module --with-pcre --http-client-body-temp-path=/var/temp/nginx/client --http-proxy-temp-path=/var/temp/nginx/proxy --http-fastcgi-temp-path=/var/temp/nginx/fastcgi --http-uwsgi-temp-path=/var/temp/nginx/uwsgi --http-scgi-temp-path=/var/temp/nginx/scgi --with-http_ssl_module --with-http_realip_module
make
make install
4. 启动nginx
/usr/local/nginx/sbin/nginx
nginx -s stop
nginx -s reload
